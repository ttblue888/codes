#include<iostream>
using namespace std;
int main()
{
	int a[100];
	int n;
	while (cin >> n)
	{
		int max = 0, min = 101;
		for (int i = 0; i < n; i++)
			cin >> a[i];
		for (int i = 0; i < n; i++)
		{
			if (a[i]<min)
				min = a[i];
			if (a[i] > max)
				max = a[i];
		}
		cout << max <<" "<< min<<endl;
	}
	system("pause");
}
